Installation
------------

::
  virtualenv --no-site-package --distribute .virtualenvs/recodoc2
  source .virtualenvs/recodoc2/bin/activate
  # Django 1.3
  pip install django
  pip install lxml
  pip install psycopg2
  pip install pylibmc
  pip install pyenchant
  pip install sphinx
  pip install Py4J
  pip install ipython
  pip install django-test-coverage
  pip install chardet
